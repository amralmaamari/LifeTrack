              CREATE PROCEDURE SP_AddNewMeasurements
(
    @MeasurementID int OUTPUT,
    @Title nvarchar(100)
)
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRY
        INSERT INTO Measurements (
            Title
        ) VALUES (
            @Title
        );
        SELECT @MeasurementID = SCOPE_IDENTITY();
    END TRY
    BEGIN CATCH
        DECLARE @ErrorMessage NVARCHAR(4000), @ErrorSeverity INT, @ErrorState INT;
        SELECT
            @ErrorMessage = ERROR_MESSAGE(),
            @ErrorSeverity = ERROR_SEVERITY(),
            @ErrorState = ERROR_STATE();
        RAISERROR (@ErrorMessage, @ErrorSeverity, @ErrorState);
        SET @MeasurementID = NULL;
    END CATCH;
END;


                GO

        CREATE PROCEDURE SP_IsMeasurementsExistsByMeasurementID
        (
            @MeasurementID int
        )
        AS
        BEGIN
            SET NOCOUNT ON;
            BEGIN TRY
                IF EXISTS (SELECT 1 FROM Measurements WHERE MeasurementID = @MeasurementID)
                BEGIN 
                    RETURN 1;
                END
                ELSE
                BEGIN
                    RETURN 0;
                END
            END TRY
            BEGIN CATCH
                DECLARE @ErrorMessage NVARCHAR(4000), @ErrorSeverity INT, @ErrorState INT;
                SELECT 
                    @ErrorMessage = ERROR_MESSAGE(),
                    @ErrorSeverity = ERROR_SEVERITY(),
                    @ErrorState = ERROR_STATE();
                RAISERROR (@ErrorMessage, @ErrorSeverity, @ErrorState);
            END CATCH;
        END;

              CREATE PROCEDURE SP_UpdateMeasurementsByID
    @MeasurementID int,
    @Title nvarchar (100)  AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRY
    UPDATE Measurements
    SET
        Title = @Title    WHERE MeasurementID = @MeasurementID;
    END TRY
    BEGIN CATCH
        SELECT
            ERROR_NUMBER() AS ErrorNumber,
            ERROR_STATE() AS ErrorState,
            ERROR_SEVERITY() AS ErrorSeverity,
            ERROR_PROCEDURE() AS ErrorProcedure,
            ERROR_LINE() AS ErrorLine,
            ERROR_MESSAGE() AS ErrorMessage;
    END CATCH;
END;
